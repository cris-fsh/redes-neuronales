Introducción

Para poder entender la capacidad del cerebro biológico para el reconocimiento perceptual, generalización, el acto de recordar, y pensar, debemos primero tener respuesta a las siguientes interrogantes:
  1. ¿Cómo es que la información acerca del mundo físico es sensada o detectada por el sistema biológico?
  2. ¿De qué forma es almacenada y recordada esta información?
  3. ¿Cómo influye esta información contenida en memoria en el posterior reconocimiento del mundo físico y el comportamiento?
La primera pregunta está casi contestada gracias al conocimiento de la fisiología de los sentidos. Sin embargo las dos últimas preguntas son objeto de especulación, y los neurofisiólogos no han integrado una teoría completa.
Existen dos posturas sobre el almacenamiento de la información en el cerebro. Una de ellas sugiere que la información sensada es almacenada en forma de representaciones codificadas o imagenes en una especie de relación uno a uno entre el estímulo sensor y el patrón codificado almacenado. Luego cada nuevo estímulo entrante involucra la búsqueda del patrón o comparación sistemática de los contenidos de la memoria con los patrones entrantes para determinar cuál será la reacción apropiada para el organismo.
La otra postura que surge del empiricismo británico, propone que no se almacena una imagen o representación codificada de los estímulos sensores como tal, y que el sistema nervioso solo funciona como un complejo sistema de interruptores, donde la retención da la forma a nuevas conexiones. Lo importante es que nunca existe un mapeo simple uno a uno entre estímulo y memoria basado en un código que permitiera su posterior reconstrucción. Cualquier información retenida debería ser almacenada como una preferencia para una particular respuesta, es decir, la información se contiene mediante conexiones o asociaciones entre estímulo y respuesta en lugar de una representación topológica.
Esta segunda postura dio pie a una teoría que representa a un sistema nervioso hipotético, o máquina, llamada perceptrón.
Para 1958 el avance de la computación, la teoría de la conmutación y la lógica simbólica permitió el desarrollo de funciones lógicas o algoritmos que emulaban el recuerdo, la comparación de estímulos, entre otras, en respuesta a secuencias de estímulos. Incluso Von Neumann y Ashby se preguntaban cómo se podría hacer para que una red neuronal imperfecta, que contiene muchas conexiones aleatorias fuera confiable para representar aquellas funciones que podrían ser bien representadas por diagramas de cableado idealizados.
El problema era que la lógica y el álgebra booleana no eran herramientas matemáticas adecuadas para tales investigaciones. Y esto fue lo que Rosenblatt resolvió en su artículo científico en términos de la teoría de la probabilidad, resolviendo varias limitaciones del modelado determinístico del cerebro.
Rossenblat partió del análisis de ciertos modelos biológicos del sistema nervioso y la primer cosa que modificó fue pasar de un modelo de trasductor continuo existente en cierta unidad llamada A-unit (es decir de una frontera de decisión basada en una función contínua), a un modelo de umbral fijo. Sin embargo, mantiene el concepto de dos tipos de conexiones neuronales: conexiones excitatorias y conexiones inhibitorias. El modelo de perceptrón que se observa en el paper es de un perceptrón óptico, y es un poco más complejo que el perceptrón que hemos estudiado hasta el momento. Incluye también un mecanismo de retroalimentación.
En esta práctica se implementará el modelo del perceptrón moderno y su correspondiente algoritmo de aprendizaje.

function [W, b] = perceptron_one_layer_t(P, t, times, tol)
%P: Entradas. Ingresar como una matriz de celdas 1xq en la que cada
%elemento es un punto de dimensión R (vector columna de longitud R). Ejemplo con q=2 y R=3: {[1; 2; 1], [-1; -2; -2]}
%t: Targets. Ingresar como una matriz de celdas 1xq en la que cada
%elemento es un target de dimensión n (vector columna de longitud n). Ejemplo con q=2 y n=1: {[0], [1]}.
%times: Número de épocas de entrenamiento.
%tol: Argumento opcional. Si se logra entrar dentro de la tolerancia
%indicada, el programa finaliza su entrenamiento antes de cumplir con
%la cantidad de épocas de entrenamiento indicada.
arguments
P (1,:) cell {same_dim(P)}
t (1,:) cell {same_dim(t), same_size(P, t)}
times uint32
tol double = 0
end
R = length(P{1}); %Dimensión de entradas.
n = length(t{1}); %Dimensión de targets, cantidad de neuronas.
l = length(P); %Cantidad de datapoints
W=zeros(n, R);
b=zeros(n, 1);
for in = 1:times
point=mod(in-1,l)+1; %Permite dar la vuelta nuevamente al primer punto.
a=hardlim(W*P{point}+b);
e=t{point}-a;
Want=W;
W=W+e*P{point}';
b=b+e;
if norm(W-Want)<tol
disp('Entrenamiento completado, se logró entrar dentro de la tolerancia indicada')
return
end
end
disp('Completado, alcanzando el número de épocas indicado')
end
function same_dim(t)
if ~all(cellfun(@(e) isequal(size(t{1}), size(e)) , t(2:end)))
eid = 'Size:notEqual';
msg = 'Todas las entradas deben tener la misma dimensión entre sí (R). Todos los targets deben de tener la misma dimensión entre sí (aunque con diferente dimensión que las entradas) (n).';
throwAsCaller(MException(eid, msg))
end
end
function same_size(a,b)
% Test for equal size
if ~isequal(size(a),size(b))
eid = 'Size:notEqual';
msg = 'La cantidad de entradas debe ser igual a la cantidad de targets (q)';
throwAsCaller(MException(eid,msg))
end
end

function [W,b] = algo_per_ite(W0,b0,p,t)
a = hardlim(W0 * p + b0);
e = t - a;
W = W0 + e * p';
b = b0 + e;
end
% verify if matrix with [W b] is in every set of a matrix with the last training [W b] values
% var_matrix = Matriz conformada por [W b] siendo las filas de W las que se multiplan por P
% check_matrix = Matriz con los valores de [W b] en la última temporada de entrenamiento
% n_neuron = numero de neuronas
% n_dimension = numero de dimensiones. Para R2 = 2, R3 = 3
function [solution] = is_solution(var_matrix, check_matrix, n_neuron, n_dimension)
% Buscamos si NO hay match entre var_matrix y los conjuntos de
% check_matrix
solution = 1;
n_row = size(check_matrix, 1);
% n_var = (# de neuronas * # de pesos para una neurona) + # de bias
n_var = n_neuron*n_dimension + n_neuron;
for i = 1:n_neuron:n_row
% values (matrix bool) = 1 si tienen el mismo valor dada una
% misma posición en ambas matrices
values = check_matrix(i:(i+n_neuron-1),:) == var_matrix;
% si el número de 1s en values no es el mismo que el número de
% variables por encontrar
if sum(sum(values)) ~= n_var
solution = 0;
end
end
end

q = .6; % Separación de las clases
K = 30; % Número de muestras en cada clase
A = [rand(1,K)-(3*q); rand(1,K)+q];
B = [rand(1,K)-q; rand(1,K)+q];
C = [rand(1,K)+q; rand(1,K)+q];
D = [rand(1,K)+(3*q); rand(1,K)+q];
E = [rand(1,K)+(3*q); rand(1,K)-q];
F = [rand(1,K)+q; rand(1,K)-q];
G = [rand(1,K)-q; rand(1,K)-q];
H = [rand(1,K)-(3*q); rand(1,K)-q];
% Salidas
a = [0 0 0 1]';
b = [0 0 1 1]';
c = [0 1 1 1]';
d = [1 1 1 1]';
e = [1 1 1 0]';
f = [0 1 1 0]';
g = [0 0 1 0]';
h = [0 0 0 0]';
% Definen las entradas (combinación para las ocho clases)
P = [A B C D E F G H];
% Definen las salidas
T = [repmat(a,1,length(A)) repmat(b,1,length(B)) ...
repmat(c,1,length(C)) repmat(d,1,length(D)) ...
repmat(e,1,length(E)) repmat(f,1,length(F)) ...
repmat(g,1,length(G)) repmat(h,1,length(H))];

function CloneFig(inFigNum,OutFigNum)
% this program copies a figure to another figure
% example: CloneFig(1,4) would copy Fig. 1 to Fig. 4
% Matt Fetterman, 2009
% pretty much taken from Matlab Technical solutions:
% http://www.mathworks.com/support/solutions/en/data/1-1UTBOL/?solution=1-1UTBOL
hf1=figure(inFigNum);
hf2=figure(OutFigNum);
clf;
compCopy(hf1,hf2);
function compCopy(op, np)
%COMPCOPY copies a figure object represented by "op" and its % descendants to another figure "np" preserving the same hierarchy.
ch = get(op, 'children');
if ~isempty(ch)
nh = copyobj(ch,np);
for k = 1:length(ch)
compCopy(ch(k),nh(k));
end
end
end
end
